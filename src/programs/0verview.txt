  module close:
  output CLOSE;

  require [OPEN] 
  ensure [OPEN].[CLOSE] 
  @*/

  end module


  module manager:
  input BTN;
  output CLOSE;

  /*@
  require [] 
  ensure ([BTN].[CLOSE]\/[])^*
  @*/

  signal OPEN in 

      emit OPEN;
      present BTN 
        then run close
        else nothing
      end present;
      pause

  end signal

  end module
