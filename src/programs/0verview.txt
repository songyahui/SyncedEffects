  module close:
  output CLOSE;

  /*@
  require [OPEN] 
  ensure [].[CLOSE] 
  @*/

  pause; 
  emit CLOSE;

  end module
  
  
  module manager:
  input BTN;
  output CLOSE;

  /*@
  require [] 
  ensure ([BTN].[CLOSE]\/[])*
  @*/

  signal OPEN in 
    loop
      emit OPEN;
      present BTN 
        then run close; 
        else nothing
      end present;
      pause
    end loop
  end signal

  end module
